# Install  

```
npm install react-parabola --save-dev
```

# Usage

```
import { showParabola } from 'react-parabola';

showParabola({
	duration: 1500,
	rate: 0.8,
	delay: 50,
	top: 80,
	start: {
		x: 0,
		y: 500,
		width: 20,
		height: 20,
	},
	end: {
		x: 500,
		y: 500,
		width: 20,
		height: 20,
	},
});
```
